204  def recieveThread(self):
205  """
206  Runs in the background, recieving bytes from the interlink node.
207 
208  When a valid message is detected, it calls the callback
209 
210  """
211  alive = True
212  message = []
213  previousByte = None
214 
215  while alive:
216  indata = self.ser.read(1)
217  if indata:
218  #check if this is the start of the message
219  if indata == self.startsyncbyte2.to_bytes(1, byteorder='big'):
220  if previousByte == self.startsyncbyte1.to_bytes(1, byteorder='big') :
221  #if so, clear the message buffer
222  message = [int.from_bytes(previousByte, byteorder='big')]
223  #check if this is the beginning of the end of the message
224  if indata == self.endsyncbyte2.to_bytes(1, byteorder='big'):
225  if previousByte == self.endsyncbyte1.to_bytes(1, byteorder='big'):
226  #if so, decode it and call the callback
227  message.append(int.from_bytes(indata, byteorder='big'))
228  package = self.decodeCANMessage(message)
229  self.recieveCallback(package)
230  # this is a data byte. append it to the package.
231  message.append(int.from_bytes(indata, byteorder='big'))
232  previousByte = indata

236  def decodeCANMessage(self, message):
237  """
238  Takes a interlink protocol message and converts it into HandyCAN
239 
240  Args:
241  message: interlink message from interlink node
242 
243  returns:
244  A HandyCAN package
245 
246  examples:
247  >>> message = [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
248  >>> hc = HandyCAN(0)
249  >>> pack = hc.decodeCANMessage(message)
250  >>> pack.length == 6
251  True
252  >>> pack.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
253  True
254  >>> pack.source == 1
255  True
256  >>> pack.dest == 2
257  True
258  >>> hc.encodeCANMessage(pack)
259  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
260 
261  """
262  pack = HandyCANPackage()
263 
264  self.log.debug(" Recieved: {}".format([hex(elem) for elem in message]))
265 
266  if message[0] != self.startsyncbyte1 or message[1] != self.startsyncbyte2:
267  pack.error = "not a valid handycan message: invalid header. Expected {} {} got {} {}".format(hex(self.startsyncbyte1), hex(self.startsyncbyte2), hex(message[0]), hex(message[1]))
268  return pack
269  if message[-1] != self.endsyncbyte2 or message[-2] != self.endsyncbyte1:
270  pack.error = "not a valid handycan message: invalid footer. Expected {} {} got {} {}".format(hex(self.endsyncbyte1), hex(self.endsyncbyte2), hex(message[-2]), hex(message[-1]))
271  return pack
272 
273  #First of all, get the data length.
274  #We need this to properly decode the package
275  pack.length = message[self.lenidx]
276  # get the total length of the package, including overhead
277 
278  #total lenght of the message is startsync + len + checklen + idl + idh + data + checkbytes + endsync
279  # or 2 + 1 + 1 + 1 + 1 + length + checklen + 2
280  # or 8 + length + checklen
281  total_length = 8 + message[self.lenidx] + message[self.checklenidx]
282 
283  if total_length != len(message):
284  pack.error = "Invalid message length: expected {} got {} ".format(total_length, len(message))
285  log.error(pack.error)
286  return pack
287 
288  # Now we need to test wether there are bytes needing decrementing
289  # and decrement them.
290  n_checkbytes = message[self.checklenidx]
291  if n_checkbytes:
292  # the offset of the checkbytes is startsync + len + checklen + idl + idh + data
293  # or 2 + 1 + 1 + 1 + 1 + length or 6 + length'
294  checkbytes_start = 6 + pack.length
295  #fetch which bytes need incrementing
296  checkbytes = [message[i + checkbytes_start] for i in range(n_checkbytes)]
297  if len (checkbytes) != n_checkbytes:
298  raise IOError ("Missing byte in checkbytes!")
299  # then increment them
300  for checkbyte in checkbytes:
301  message[checkbyte] -= 1
302 
303  StdId = message[self.StdIdLidx] | message[self.StdIdHidx] << 8;
304  pack.source = (StdId & 0x3E0) >> 5;
305  pack.dest = StdId & 0x1F
306  pack.data = []
307  for i in range(self.dataidx, pack.length + 6, 1):
308  pack.data.append(message[i])
309 
310  return pack

313  def encodeCANMessage(self, pack):
314  """
315  Takes a HandyCANPackage and converts it into a interlink message
316 
317  Args:
318  pack: HandyCANPackage to convert
319 
320  Returns:
321  Interlink message
322 
323 
324  Examples:
325  >>> hcp = HandyCANPackage()
326  >>> hcp.length = 6
327  >>> hcp.data = [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
328  >>> hcp.source = 1
329  >>> hcp.dest = 2
330  >>> hc = HandyCAN(0)
331  >>> hc.encodeCANMessage(hcp)
332  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
333  >>> msg = hc.encodeCANMessage(hcp)
334  >>> hcp2 = hc.decodeCANMessage(msg)
335  >>> hcp.length == 6
336  True
337  >>> hcp.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
338  True
339  >>> hcp.source == 1
340  True
341  >>> hcp.dest == 2
342  True
343 
344  """
345  StdId = pack.dest | (pack.source << self.sourceoffset)
346  message = [
347  #0, 1 start sync
348  self.startsyncbyte1, self.startsyncbyte2,
349  #2 length
350  pack.length,
351  #3: number of checkbytes
352  0,
353  #4: stdid low
354  StdId & 0xFF,
355  #5: stdid high
356  ((StdId & 0xFF00) >> 8) & 0xFF,
357  ]
358  dataOffset = len(message)
359  # todo: check if length <= length of list pack.data
360  for i in range(pack.length):
361  message.append(pack.data[i])
362 
363  shift = []
364  #Generate sync
365  for check in range(3, len(message), 1):
366  if (message[check - 1] == self.startsyncbyte1 and message[check] == self.startsyncbyte2) or (message[check - 1] == self.endsyncbyte1 and message[check] == self.endsyncbyte2):
367  message[check] += 1
368  shift.append(check)
369 
370  for checkloc in shift:
371  message.append(checkloc)
372 
373  message[3]= len(shift)
374 
375  message.append(self.endsyncbyte1)
376  message.append(self.endsyncbyte2)
377 
378  return message
