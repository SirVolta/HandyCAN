207  def recieveThread(self):
208  """
209  Runs in the background, recieving bytes from the interlink node.
210 
211  When a valid message is detected, it calls the callback
212 
213  """
214  alive = True
215  message = []
216  previousByte = None
217 
218  while alive:
219  indata = self.ser.read(1)
220  if indata:
221  #check if this is the start of the message
222  if indata == self.startsyncbyte2.to_bytes(1, byteorder='big'):
223  if previousByte == self.startsyncbyte1.to_bytes(1, byteorder='big') :
224  #if so, clear the message buffer
225  message = [int.from_bytes(previousByte, byteorder='big')]
226  #check if this is the beginning of the end of the message
227  if indata == self.endsyncbyte2.to_bytes(1, byteorder='big'):
228  if previousByte == self.endsyncbyte1.to_bytes(1, byteorder='big'):
229  #if so, decode it and call the callback
230  message.append(int.from_bytes(indata, byteorder='big'))
231  package = self.decodeCANMessage(message)
232  self.recieveCallback(package)
233  # this is a data byte. append it to the package.
234  message.append(int.from_bytes(indata, byteorder='big'))
235  previousByte = indata

239  def decodeCANMessage(self, message):
240  """
241  Takes a interlink protocol message and converts it into HandyCAN
242 
243  Args:
244  message: interlink message from interlink node
245 
246  returns:
247  A HandyCAN package
248 
249  examples:
250  >>> message = [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
251  >>> hc = HandyCAN(0)
252  >>> pack = hc.decodeCANMessage(message)
253  >>> pack.length == 6
254  True
255  >>> pack.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
256  True
257  >>> pack.source == 1
258  True
259  >>> pack.dest == 2
260  True
261  >>> hc.encodeCANMessage(pack)
262  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
263 
264  """
265  pack = HandyCANPackage()
266 
267  self.log.debug(" Recieved: {}".format([hex(elem) for elem in message]))
268 
269  if message[0] != self.startsyncbyte1 or message[1] != self.startsyncbyte2:
270  pack.error = "not a valid handycan message: invalid header. Expected {} {} got {} {}".format(hex(self.startsyncbyte1), hex(self.startsyncbyte2), hex(message[0]), hex(message[1]))
271  return pack
272  if message[-1] != self.endsyncbyte2 or message[-2] != self.endsyncbyte1:
273  pack.error = "not a valid handycan message: invalid footer. Expected {} {} got {} {}".format(hex(self.endsyncbyte1), hex(self.endsyncbyte2), hex(message[-2]), hex(message[-1]))
274  return pack
275 
276  #First of all, get the data length.
277  #We need this to properly decode the package
278  pack.length = message[self.lenidx]
279  # get the total length of the package, including overhead
280 
281  #total lenght of the message is startsync + len + checklen + idl + idh + data + checkbytes + endsync
282  # or 2 + 1 + 1 + 1 + 1 + length + checklen + 2
283  # or 8 + length + checklen
284  total_length = 8 + message[self.lenidx] + message[self.checklenidx]
285 
286  if total_length != len(message):
287  pack.error = "Invalid message length: expected {} got {} ".format(total_length, len(message))
288  log.error(pack.error)
289  return pack
290 
291  # Now we need to test wether there are bytes needing decrementing
292  # and decrement them.
293  n_checkbytes = message[self.checklenidx]
294  if n_checkbytes:
295  # the offset of the checkbytes is startsync + len + checklen + idl + idh + data
296  # or 2 + 1 + 1 + 1 + 1 + length or 6 + length'
297  checkbytes_start = 6 + pack.length
298  #fetch which bytes need incrementing
299  checkbytes = [message[i + checkbytes_start] for i in range(n_checkbytes)]
300  if len (checkbytes) != n_checkbytes:
301  raise IOError ("Missing byte in checkbytes!")
302  # then increment them
303  for checkbyte in checkbytes:
304  message[checkbyte] -= 1
305 
306  StdId = message[self.StdIdLidx] | message[self.StdIdHidx] << 8;
307  pack.source = (StdId & 0x3E0) >> 5;
308  pack.dest = StdId & 0x1F
309  pack.data = []
310  for i in range(self.dataidx, pack.length + 6, 1):
311  pack.data.append(message[i])
312 
313  return pack

316  def encodeCANMessage(self, pack):
317  """
318  Takes a HandyCANPackage and converts it into a interlink message
319 
320  Args:
321  pack: HandyCANPackage to convert
322 
323  Returns:
324  Interlink message
325 
326 
327  Examples:
328  >>> hcp = HandyCANPackage()
329  >>> hcp.length = 6
330  >>> hcp.data = [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
331  >>> hcp.source = 1
332  >>> hcp.dest = 2
333  >>> hc = HandyCAN(0)
334  >>> hc.encodeCANMessage(hcp)
335  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
336  >>> msg = hc.encodeCANMessage(hcp)
337  >>> hcp2 = hc.decodeCANMessage(msg)
338  >>> hcp.length == 6
339  True
340  >>> hcp.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
341  True
342  >>> hcp.source == 1
343  True
344  >>> hcp.dest == 2
345  True
346 
347  """
348  StdId = pack.dest | (pack.source << self.sourceoffset)
349  message = [
350  #0, 1 start sync
351  self.startsyncbyte1, self.startsyncbyte2,
352  #2 length
353  pack.length,
354  #3: number of checkbytes
355  0,
356  #4: stdid low
357  StdId & 0xFF,
358  #5: stdid high
359  ((StdId & 0xFF00) >> 8) & 0xFF,
360  ]
361  dataOffset = len(message)
362  # todo: check if length <= length of list pack.data
363  for i in range(pack.length):
364  message.append(pack.data[i])
365 
366  shift = []
367  #Generate sync
368  for check in range(3, len(message), 1):
369  if (message[check - 1] == self.startsyncbyte1 and message[check] == self.startsyncbyte2) or (message[check - 1] == self.endsyncbyte1 and message[check] == self.endsyncbyte2):
370  message[check] += 1
371  shift.append(check)
372 
373  for checkloc in shift:
374  message.append(checkloc)
375 
376  message[3]= len(shift)
377 
378  message.append(self.endsyncbyte1)
379  message.append(self.endsyncbyte2)
380 
381  return message
