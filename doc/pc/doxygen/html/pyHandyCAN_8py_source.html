171  def transmitThread(self):
172  while True:
173  if not self.cts:
174  time.sleep(0.000001) # short cpu-unload delay
175  #cts is not used, wait for the minimum time between packs
176  timediff = datetime.datetime.now() - self.last_package_sent_time
177  self.log.debug("waiting to transmit")
178  if timediff.microseconds > self.minimum_time_between_packages:
179  message = self.transmitQueue.get(block=True, timeout=None)
180  self.log.debug("Delay sending: {} ".format([hex(elem) for elem in message]))
181  self.ser.write(message)
182  self.last_package_sent_time = datetime.datetime.now()
183  self.transmitQueue.task_done()
184  else:
185  #wait for 200 microseconds for cts line delay
186  #maximum speed: 4000 packages per second
187  time.sleep(200.0 * (10.0 ** -6))
188  if self.txe:
189  if self.ser.cts:
190  message = self.transmitQueue.get(block=True, timeout=None)
191  self.log.debug("CTS Sending: {} ".format([hex(elem) for elem in message]))
192  bytes_sent = self.ser.write(message)
193  self.last_package_sent_time = datetime.datetime.now()
194  self.transmitQueue.task_done()

248  def recieveThread(self):
249  """
250  Runs in the background, recieving bytes from the interlink node.
251 
252  When a valid message is detected, it calls the callback
253 
254  """
255  alive = True
256  message = []
257  previousByte = None

261  if indata:

264  if previousByte == self.startsyncbyte1.to_bytes(1, byteorder='big') :
265  #if so, clear the message buffer

268  if indata == self.endsyncbyte2.to_bytes(1, byteorder='big'):
269  if previousByte == self.endsyncbyte1.to_bytes(1, byteorder='big'):
270  #if so, place it into the queue
271  message.append(int.from_bytes(indata, byteorder='big'))
272  self.recieveQueue.put(list(message))
273  # this is a data byte. append it to the package.
274  message.append(int.from_bytes(indata, byteorder='big'))
275  previousByte = indata

288  def decodeCANMessage(self, message):
289  """
290  Takes a interlink protocol message and converts it into HandyCAN
291 
292  Args:
293  message: interlink message from interlink node
294 
295  returns:
296  A HandyCAN package
297 
298  examples:
299  >>> message = [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
300  >>> hc = HandyCAN(0)
301  >>> pack = hc.decodeCANMessage(message)
302  >>> pack.length == 6
303  True
304  >>> pack.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
305  True
306  >>> pack.source == 1
307  True
308  >>> pack.dest == 2
309  True
310  >>> hc.encodeCANMessage(pack)
311  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
312 
313  """
314  pack = HandyCANPackage()
315 
316  self.log.debug("Decoding: {}".format([hex(elem) for elem in message]))
317 
318  if message[0] != self.startsyncbyte1 or message[1] != self.startsyncbyte2:
319  pack.error = "not a valid handycan message: invalid header. Expected {} {} got {} {}".format(hex(self.startsyncbyte1), hex(self.startsyncbyte2), hex(message[0]), hex(message[1]))
320  self.log.error(pack.error)
321  return pack
322  if message[-1] != self.endsyncbyte2 or message[-2] != self.endsyncbyte1:
323  pack.error = "not a valid handycan message: invalid footer. Expected {} {} got {} {}".format(hex(self.endsyncbyte1), hex(self.endsyncbyte2), hex(message[-2]), hex(message[-1]))
324  self.log.error(pack.error)
325  return pack
326 
327  #First of all, get the data length.
328  #We need this to properly decode the package
329  pack.length = message[self.lenidx]
330  # get the total length of the package, including overhead
331 
332  #total lenght of the message is startsync + len + checklen + idl + idh + data + checkbytes + endsync
333  # or 2 + 1 + 1 + 1 + 1 + length + checklen + 2
334  # or 8 + length + checklen
335  total_length = 8 + message[self.lenidx] + message[self.checklenidx]
336 
337  if total_length != len(message):
338  pack.error = "Invalid message length: expected {} got {} ".format(total_length, len(message))
339  self.log.error(pack.error)
340  return pack
341 
342  # Now we need to test wether there are bytes needing decrementing
343  # and decrement them.
344  n_checkbytes = message[self.checklenidx]
345  if n_checkbytes:
346  # the offset of the checkbytes is startsync + len + checklen + idl + idh + data
347  # or 2 + 1 + 1 + 1 + 1 + length or 6 + length'
348  checkbytes_start = 6 + pack.length
349  #fetch which bytes need incrementing
350  checkbytes = [message[i + checkbytes_start] for i in range(n_checkbytes)]
351  if len (checkbytes) != n_checkbytes:
352  raise IOError ("Missing byte in checkbytes!")
353  # then increment them
354  for checkbyte in checkbytes:
355  message[checkbyte] -= 1
356 
357  StdId = message[self.StdIdLidx] | message[self.StdIdHidx] << 8;
358  pack.source = (StdId & 0x3E0) >> 5;
359  pack.dest = StdId & 0x1F
360  pack.data = []
361  for i in range(self.dataidx, pack.length + 6, 1):
362  pack.data.append(message[i])
363 
364  return pack

367  def encodeCANMessage(self, pack):
368  """
369  Takes a HandyCANPackage and converts it into a interlink message
370 
371  Args:
372  pack: HandyCANPackage to convert
373 
374  Returns:
375  Interlink message
376 
377 
378  Examples:
379  >>> hcp = HandyCANPackage()
380  >>> hcp.length = 6
381  >>> hcp.data = [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
382  >>> hcp.source = 1
383  >>> hcp.dest = 2
384  >>> hc = HandyCAN(0)
385  >>> hc.encodeCANMessage(hcp)
386  [240, 250, 6, 2, 34, 0, 0, 240, 251, 0, 224, 239, 8, 11, 224, 238]
387  >>> msg = hc.encodeCANMessage(hcp)
388  >>> hcp2 = hc.decodeCANMessage(msg)
389  >>> hcp.length == 6
390  True
391  >>> hcp.data == [0, 0xF0, 0xFA, 0, 0xE0, 0xEE]
392  True
393  >>> hcp.source == 1
394  True
395  >>> hcp.dest == 2
396  True
397 
398  """
399  StdId = pack.dest | (pack.source << self.sourceoffset)
400  message = [
401  #0, 1 start sync
402  self.startsyncbyte1, self.startsyncbyte2,
403  #2 length
404  pack.length,
405  #3: number of checkbytes
406  0,
407  #4: stdid low
408  StdId & 0xFF,
409  #5: stdid high
410  ((StdId & 0xFF00) >> 8) & 0xFF,
411  ]
412  dataOffset = len(message)
413  # todo: check if length <= length of list pack.data
414  for i in range(pack.length):
415  message.append(pack.data[i])
416 
417  shift = []
418  #Generate sync
419  for check in range(3, len(message), 1):
420  if (message[check - 1] == self.startsyncbyte1 and message[check] == self.startsyncbyte2) or (message[check - 1] == self.endsyncbyte1 and message[check] == self.endsyncbyte2):
421  message[check] += 1
422  shift.append(check)
423 
424  for checkloc in shift:
425  message.append(checkloc)
426 
427  message[3]= len(shift)
428 
429  message.append(self.endsyncbyte1)
430  message.append(self.endsyncbyte2)
431 
432  return message
